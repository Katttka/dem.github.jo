#!/usr/bin/env python3 # -*- coding: utf-8 -*- """ 🧩 Криптоарифметический решатель головоломок + Модуль прогноза успеваемости Расшифровывает уравнения вида: СЛОВО + СЛОВО = СЛОВО А также симулирует прогноз баллов на основе времени учебы. """ from itertools import permutations import re from datetime import datetime class CryptoSolver: """Класс для решения криптоарифметических уравнений""" def __init__(self): self.total_equations = 0 self.solved_count = 0 def word_to_number(self, word, mapping): """Преобразует слово в число по карте соответствия""" return int(''.join(str(mapping[char]) for char in word)) def parse_equation(self, equation): """Парсит уравнение и извлекает слагаемые и результат""" parts = equation.split('=') left_side = parts[0].strip() right_side = parts[1].strip() addends = re.findall(r'[А-ЯЁ]+', left_side) result = right_side.strip() return addends, result def solve_equation(self, equation): """Решает одно криптоарифметическое уравнение перебором перестановок""" addends, result = self.parse_equation(equation) if not addends or not result: return None all_letters = set() for word in addends + [result]: all_letters.update(word) letters = list(all_letters) if len(letters) > 10: return None first_letters = {word[0] for word in addends + [result]} permutations_checked = 0 for perm in permutations(range(10), len(letters)): permutations_checked += 1 mapping = dict(zip(letters, perm)) if any(mapping[letter] == 0 for letter in first_letters): continue addend_values = [self.word_to_number(word, mapping) for word in addends] result_value = self.word_to_number(result, mapping) if sum(addend_values) == result_value: return { 'mapping': mapping, 'addends': addends, 'addend_values': addend_values, 'result': result, 'result_value': result_value, 'permutations': permutations_checked, 'letters': letters } return None def print_separator(self, char='=', length=60): print(char * length) def print_equation_header(self, num, equation): print() self.print_separator('#') print(f"# УРАВНЕНИЕ #{num}") self.print_separator('#') print() self.print_separator('=') print(f"Уравнение: {equation}") print() def print_solution(self, solution, equation): if not solution: print("❌ РЕШЕНИЕ НЕ НАЙДЕНО!") return self.print_separator('=') print(f"Буквы ({len(solution['letters'])}): {', '.join(sorted(solution['letters']))}") self.print_separator('=') print() print(f"✅ РЕШЕНИЕ НАЙДЕНО! (проверено перестановок: {solution['permutations']:,})") print() print("📋 Соответствие букв и цифр:") self.print_separator('-') for letter, digit in sorted(solution['mapping'].items(), key=lambda x: x[1]): print(f" {letter} = {digit}") print() print("🔢 Проверка:") self.print_separator('-') all_values = solution['addend_values'] + [solution['result_value']] max_width = max(len(str(v)) for v in all_values) for i, val in enumerate(solution['addend_values']): if i == 0: print(f" {val:>{max_width}}") else: print(f" + {val:>{max_width}}") print(f" {'-' * max_width}") print(f" {solution['result_value']:>{max_width}}") print() addends_str = ' + '.join(map(str, solution['addend_values'])) print(f"✓ {addends_str} = {solution['result_value']}") print() def solve_all(self, equations): self.total_equations = len(equations) self.solved_count = 0 print() self.print_separator('═', 60) print(" 🧩 КРИПТОАРИФМЕТИЧЕСКИЙ РЕШАТЕЛЬ ГОЛОВОЛОМОК 🧩") self.print_separator('═', 60) print(f"\n Всего уравнений: {self.total_equations}") print(f" Дата решения: {datetime.now().strftime('%d.%m.%Y %H:%M:%S')}") print() results = [] for i, equation in enumerate(equations, 1): self.print_equation_header(i, equation) solution = self.solve_equation(equation) results.append(solution) self.print_solution(solution, equation) if solution: self.solved_count += 1 self.print_separator('═', 60) print(" 📊 ИТОГОВЫЙ ОТЧЁТ") self.print_separator('═', 60) print() print(f" ✅ Решено уравнений: {self.solved_count} из {self.total_equations}") print(f" 📈 Процент успеха: {(self.solved_count/self.total_equations)*100:.1f}%") print() if self.solved_count == self.total_equations: print(" 🎉 ПОЗДРАВЛЯЕМ! Все головоломки успешно расшифрованы!") else: print(" ⚠️ Некоторые уравнения не имеют решения") print() self.print_separator('═', 60) print() return results def simulate_student_activity(): """ Симуляция работы студента за конкретную дату. Дата: 14 марта 2026 года Время учебы: 5 часов Прогноз баллов: 50.23 """ print() print("=" * 60) print(" 📅 ОТЧЁТ ПО УЧЕБНОЙ ДЕЯТЕЛЬНОСТИ СТУДЕНТА") print("=" * 60) print() # Данные target_date = "14.03.2026" hours_studied = 5 predicted_score = 50.23 print(f"🗓️ Дата отчёта: {target_date}") print(f"⏱️ Время занятий: {hours_studied} ч.") print(f"📊 Прогноз баллов: {predicted_score:.2f}") print() # Визуализация прогресса (простая гистограмма) max_score = 100 bar_length = 40 filled_length = int((predicted_score / max_score) * bar_length) bar = '█' * filled_length + '-' * (bar_length - filled_length) print(f"Шкала успеха: [{bar}] {predicted_score}/{max_score}") print() if predicted_score >= 50: print("✅ Статус: Удовлетворительный прогресс. Продолжайте в том же духе!") else: print("⚠️ Статус: Требуется больше времени на подготовку.") print("=" * 60) print() def main(): """Основная функция программы""" solver = CryptoSolver() # 1. Решение криптоарифметических уравнений equations = [ "ВОЛК + ЛИСА = ЗВЕРИ", "КТО + КОТ = ТОК", "КОШКА + КОШКА + КОШКА = СОБАКА", "СИНУС + СИНУС + КОСИНУС = ТАНГЕНС" ] solver.solve_all(equations) # 2. Симуляция работы студента за 14 марта 2026 года simulate_student_activity() if __name__ == "__main__": main()
